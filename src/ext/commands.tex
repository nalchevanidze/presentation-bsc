\newcommand{\importGQL}[2]{
    \lstinputlisting[
        caption={#2},
    ]{src/code/#1.gql}   
}

\newcommand{\importHS}[2]{
    \lstinputlisting[
        language={haskell},
        caption={#2},
    ]{src/code/#1.hs}   
}

\newcommand{\importHSFragment}[4]{
    \lstinputlisting[
        language={haskell},
        caption={#4},
        firstline=#2, 
        lastline=#3,
    ]{src/code/#1.hs}   
}
